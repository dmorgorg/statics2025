      \draw[latex-latex] ($(A)+(-0.75,0)$)--node[fill=white]{$1900\,$mm}($(D)+(-8.75,0)$);
      \draw[latex-latex] ($(A)+(0,-0.75)$)--node[fill=white]{$2420\,$mm}($(C)+(0,-0.75)$);
      
      \draw[ultra thick,-Latex] (D)--+(-2,0)node[yshift=0.325cm,black]{$3.75\,$kN};

      \fill (A) circle (2pt)node[xshift=-0.25cm, yshift=0.25cm]{ $A$};
      \fill (C) circle (2pt)node[xshift=-0.25cm, yshift=0.25cm]{ $C$};
      \fill (D) circle (2pt)node[yshift=0.325cm]{ $D$};

      \draw[very thick, -latex] (C)--+(90:1.5)node[left]{$ R_{Cy} $};
      \draw[very thick, -latex] (C)--+(0:1.5)node[above]{$ R_{Cx} $};
      \draw[very thick, -latex] (A)--+(90:1.5)node[right]{$ R_{Ay} $};
    }
    \begin{align*}
      \Sigma M_C &= (3.75\,\textsf{kN})\cd(1.900\,\textsf{m})-R_{Ay}\cd (2.420\,\textsf{m})=0\\[0.25em]
      \Ra R_{Ay} &= 2.9442\,\mathsf{kN\cd m} \\\\
      \Sigma F_x &= R_{Cx}-3.75\,\textsf{kN} = 0 \\[0.25em]
      \Ra R_{Cx} &= 3.75\,\textsf{kN}\\\\
      \Sigma F_y &= R_{Cy}+2.9442\,\textsf{kN} = 0 \\[0.25em]
      \Ra R_{Cy} &= -2.9442\,\textsf{kN}
    \end{align*}
	}
  \cbox{
    \underline{Note:}
    \begin{enumerate}
      \item $BE$ is a zero-force member
      \item Since, $BE$ is a zero force member, so is $BD$
      \item Thus, $F_{AE}=F_{ED}$ and $F_{AB}=F_{BC}$
    \end{enumerate}
  }

    \underline{Joint A:}\parm\centering

  }
