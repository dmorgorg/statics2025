      \underline{\bf Draw Section $\bm{a\mm a}$ and analyze the left portion of the truss.}\parb
      \mini[0.325]{
        \centering
        \def\scale{0.5}
        \input{../../pikz/08MoS/08MoS07bHandout.tex}
        \parb
        \cbox{
          \underline{\bf Answers:}
          \begin{align*}
            F_{BC} &= 22.5\,\textsf{kN}\qquad\textsf{(Compression)}\\
            F_{BG} &= 8.39\,\textsf{kN}\qquad\textsf{(Compression)}\\
            F_{GH} &= 22.0\,\textsf{kN}\qquad\textsf{(Tension)}\\
          \end{align*}
        }
      }
      \hfill
      \mini[0.55]{      
        \begin{align*}
          \theta &= \tan^{-1}\left[\frac{4.5}{2.25}\right]=63.435\deg\\[0.25em]
          \phi &= \tan^{-1}\left[\frac{1.5}{2.25}\right]=33.690\deg          
        \end{align*}
        Take moments about $G$ to find $F_{BC}$: 
        \begin{align*}
          \Sigma M_G &= (F_{BC}\cd \cos\phi)\cd (3.00\,\textsf{m})+(5.00\,\textsf{kN})\cd (2.25\,\textsf{m})\\[0.25em]
          &\qquad\qquad +(10.00\,\textsf{kN})\cd (4.50\,\textsf{m})\\[0.25em]
          &= F_{BC}\cd (2.4962\,\textsf{m})+56.250\,\mathsf{kN\cd m}=0\\\\
          \Ra F_{BC} &= -22.534\,\textsf{kN}        
        \end{align*} 
        Take moments about $B$ to find $F_{GH}$: 
        \begin{align*}
          \Sigma M_B &= -(F_{GH})\cd (4.50\,\textsf{m})+(5.00\,\textsf{kN})\cd (2.25\,\textsf{m})\\[0.25em]
          &\qquad\qquad +(5.00\,\textsf{kN})\cd (4.50\,\textsf{m})+(10.00\,\textsf{kN})\cd (6.75\,\textsf{m})=0\\\\
          \Ra F_{GH} &= 22.500\,\textsf{kN}        
        \end{align*}
        Multiple options: sum $x$-components, sum $y$-components or take moments around point $J$ to find $F_{BG}$. (Taking moments here.)
        \begin{align*}
          \Sigma M_J &= -(F_{BG}\cd \sin\theta)\cd (4.50\,\textsf{m})-(5.00\,\textsf{kN})\cd (4.50\,\textsf{m})\\
          &\qquad\qquad -(5.00\,\textsf{kN})\cd (2.25\,\textsf{m})\\[0.25em]
          \Sigma M_J &= -F_{BG}\cd (4.0249\,\textsf{m})-(33.750\,\mathsf{kN\cd m})=0\\\\
          \Ra F_{BG} &= -8.3853\,\textsf{kN}        
        \end{align*}
      } 
    }   
  \end{textblock*}
