    \draw ($ (B)+(-68.2:1.75)  $) -- ($ (B)+(-68.2:2.75)  $);
    \draw[latex-latex] ($ (B)+(-68.2:2.2)  $) arc (-68.2:-120:2.2);

    \path[name path=AC] (A)--(150:4.75);
    \path[draw, name path=BC] (B) -- +(-120:3.75) ;

  \end{align*}
