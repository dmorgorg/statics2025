		$$ A = \frac{(\theta-\sin\theta)D^2}{8} $$

			A_{Gold} &= \frac{\left(\frac{\pi}{3}-\sin 60\deg\right)(24)^2}{8} \\

		\begin{align*}
			A_{Green+Gold} &= \frac{\left(\frac{2\pi}{3}-\sin 120\deg\right)(24)^2}{8} \\
				&= 88.443
		\end{align*}

		$$ A_{Green} = 88.443-13.044 = 75.399 $$
